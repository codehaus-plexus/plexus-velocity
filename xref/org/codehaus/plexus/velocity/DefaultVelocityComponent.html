19  import java.util.Enumeration;
20  import java.util.Properties;
21  
22  import org.apache.velocity.app.VelocityEngine;
23  import org.apache.velocity.runtime.RuntimeServices;
24  import org.apache.velocity.runtime.RuntimeConstants;
25  import org.apache.velocity.runtime.log.LogSystem;
26  import org.codehaus.plexus.personality.plexus.lifecycle.phase.Initializable;
27  import org.codehaus.plexus.personality.plexus.lifecycle.phase.InitializationException;
28  import org.codehaus.plexus.logging.AbstractLogEnabled;
29  
30  /**
31   * A simple velocity component implementation.
32   * <p/>
33   * A typical configuration will look like this:
34   * <pre>
35   *      <configuration>
36   *        <properties>
37   *          <property>
38   *            <name>runtime.log.logsystem.class</name>
39   *            <value>org.codehaus.plexus.velocity.Log4JLoggingSystem</value>
40   *          </property>
41   *          <property>
42   *            <name>runtime.log.logsystem.log4j.category</name>
43   *            <value>org.codehaus.plexus.velocity.DefaultVelocityComponentTest</value>
44   *          </property>
45   *          <property>
46   *            <name>resource.loader</name>
47   *            <value>classpath</value>
48   *          </property>
49   *          <property>
50   *            <name>classpath.resource.loader.class</name>
51   *            <value>org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader</value>
52   *          </property>
53   *        </properties>
54   *      </configuration>
55   * </pre>
56   */
57  public class DefaultVelocityComponent
58      extends AbstractLogEnabled
59      implements VelocityComponent, Initializable, LogSystem
60  {
61      private VelocityEngine engine;
62  
63      private Properties properties;
64  
65      // ----------------------------------------------------------------------
66      //
67      // ----------------------------------------------------------------------
68      
69      public void initialize()
70          throws InitializationException
71      {
72          engine = new VelocityEngine();
73  
74          // avoid "unable to find resource 'VM_global_library.vm' in any resource loader."
75          engine.setProperty( "velocimacro.library", "" );
76  
77          engine.setProperty( RuntimeConstants.RUNTIME_LOG_LOGSYSTEM, this );
78  
79          if ( properties != null )
80          {
81              for ( Enumeration e = properties.propertyNames(); e.hasMoreElements(); )
82              {
83                  String key = e.nextElement().toString();
84      
85                  String value = properties.getProperty( key );
86      
87                  engine.setProperty( key, value );
88  
89                  getLogger().debug( "Setting property: " + key + " => '" + value + "'." );
90              }
91          }
92  
93          try
94          {
95              engine.init();
96          }
97          catch ( Exception e )
98          {
99              throw new InitializationException( "Cannot start the velocity engine: ", e );
100         }
101     }
102 
103     // ----------------------------------------------------------------------
104     //
105     // ----------------------------------------------------------------------
106 
107     public VelocityEngine getEngine()
108     {
109         return engine;
110     }
111     // ----------------------------------------------------------------------
112     //
113     // ----------------------------------------------------------------------
114 
115     private RuntimeServices runtimeServices;
116 
117     public void init( RuntimeServices runtimeServices )
118     {
119         this.runtimeServices = runtimeServices;
120     }
121 
122     public void logVelocityMessage( int level, String message )
123     {
124         switch ( level )
125         {
126             case LogSystem.WARN_ID:
127                 getLogger().warn( message );
128                 break;
129             case LogSystem.INFO_ID:
130                 // velocity info messages are too verbose, just consider them as debug messages...
131                 getLogger().debug( message );
132                 break;
133             case LogSystem.DEBUG_ID:
134                 getLogger().debug( message );
135                 break;
136             case LogSystem.ERROR_ID:
137                 getLogger().error( message );
138                 break;
139             default:
140                 getLogger().debug( message );
141                 break;
142         }
143     }
144 }
